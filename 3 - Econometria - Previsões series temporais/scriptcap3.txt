IC_MMS <- function(z,r,l){
smadf <- SMA(z,r)
IC_I <- rep(smadf[length(z)],l) - 1.96*sd(z)/sqrt(r)
IC_S <- rep(smadf[length(z)],l) + 1.96*sd(z)/sqrt(r)
previsao <- rep(smadf[length(z)],l)
cbind(IC_I,previsao,IC_S)
